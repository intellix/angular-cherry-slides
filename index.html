Angular is Data-Driven pt.1

don't have to update your DOM with new data

don't have to mess about with selectors


Before... 

<table id="transactionsTable">
  <tbody>
    <?php foreach ($transactions as $transaction) { ?>
      <tr>
        <td class="amount"><?php echo $transaction['amount'] ?></td>
      </tr>
    <?php } ?>
  </tbody>
</table>

<script>
  $.ajax({
    url: '/transactions.php',
  }).done(function(data) {
    $('#transactionTable tr').remove();
    $.each(data.transactions function(transaction) {
      $('#transactionTable tbody').append('<tr><td class="amount">' + transaction + '</td></tr>');
    });
  });
</script>
